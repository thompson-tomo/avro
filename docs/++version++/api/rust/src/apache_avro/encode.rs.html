32/// Encode a `Value` into avro format.
33///
34/// **NOTE** This will not perform schema validation. The value is assumed to
35/// be valid with regards to the schema. Schema are needed only to guide the
36/// encoding for complex type values.
37pub fn encode<W: Write>(value: &Value, schema: &Schema, writer: &mut W) -> AvroResult<usize> {
38    let rs = ResolvedSchema::try_from(schema)?;
39    encode_internal(value, schema, rs.get_names(), &None, writer)
40}
41
42pub(crate) fn encode_bytes<B: AsRef<[u8]> + ?Sized, W: Write>(
43    s: &B,
44    mut writer: W,
45) -> AvroResult<usize> {
46    let bytes = s.as_ref();
47    encode_long(bytes.len() as i64, &mut writer)?;
48    writer.write(bytes).map_err(Error::WriteBytes)
49}
50
51pub(crate) fn encode_long<W: Write>(i: i64, writer: W) -> AvroResult<usize> {
52    zig_i64(i, writer)
53}
54
55pub(crate) fn encode_int<W: Write>(i: i32, writer: W) -> AvroResult<usize> {
56    zig_i32(i, writer)
57}

59pub(crate) fn encode_internal<W: Write, S: Borrow<Schema>>(
60    value: &Value,
61    schema: &Schema,
62    names: &HashMap<Name, S>,
63    enclosing_namespace: &Namespace,
64    writer: &mut W,
65) -> AvroResult<usize> {
66    if let Schema::Ref { ref name } = schema {
67        let fully_qualified_name = name.fully_qualified_name(enclosing_namespace);
68        let resolved = names
69            .get(&fully_qualified_name)
70            .ok_or(Error::SchemaResolutionError(fully_qualified_name))?;
71        return encode_internal(value, resolved.borrow(), names, enclosing_namespace, writer);
72    }
73
74    match value {
75        Value::Null => {
76            if let Schema::Union(union) = schema {
77                match union.schemas.iter().position(|sch| *sch == Schema::Null) {
78                    None => Err(Error::EncodeValueAsSchemaError {
79                        value_kind: ValueKind::Null,
80                        supported_schema: vec![SchemaKind::Null, SchemaKind::Union],
81                    }),
82                    Some(p) => encode_long(p as i64, writer),
83                }
84            } else {
85                Ok(0)
86            }
87        }
88        Value::Boolean(b) => writer.write(&[u8::from(*b)]).map_err(Error::WriteBytes),
89        // Pattern | Pattern here to signify that these _must_ have the same encoding.
90        Value::Int(i) | Value::Date(i) | Value::TimeMillis(i) => encode_int(*i, writer),
91        Value::Long(i)
92        | Value::TimestampMillis(i)
93        | Value::TimestampMicros(i)
94        | Value::TimestampNanos(i)
95        | Value::LocalTimestampMillis(i)
96        | Value::LocalTimestampMicros(i)
97        | Value::LocalTimestampNanos(i)
98        | Value::TimeMicros(i) => encode_long(*i, writer),
99        Value::Float(x) => writer.write(&x.to_le_bytes()).map_err(Error::WriteBytes),
100        Value::Double(x) => writer.write(&x.to_le_bytes()).map_err(Error::WriteBytes),
101        Value::Decimal(decimal) => match schema {
102            Schema::Decimal(DecimalSchema { inner, .. }) => match *inner.clone() {
103                Schema::Fixed(FixedSchema { size, .. }) => {
104                    let bytes = decimal.to_sign_extended_bytes_with_len(size).unwrap();
105                    let num_bytes = bytes.len();
106                    if num_bytes != size {
107                        return Err(Error::EncodeDecimalAsFixedError(num_bytes, size));
108                    }
109                    encode(&Value::Fixed(size, bytes), inner, writer)
110                }
111                Schema::Bytes => encode(&Value::Bytes(decimal.try_into()?), inner, writer),
112                _ => Err(Error::ResolveDecimalSchema(SchemaKind::from(
113                    *inner.clone(),
114                ))),
115            },
116            _ => Err(Error::EncodeValueAsSchemaError {
117                value_kind: ValueKind::Decimal,
118                supported_schema: vec![SchemaKind::Decimal],
119            }),
120        },
121        &Value::Duration(duration) => {
122            let slice: [u8; 12] = duration.into();
123            writer.write(&slice).map_err(Error::WriteBytes)
124        }
125        Value::Uuid(uuid) => match *schema {
126            Schema::Uuid | Schema::String => encode_bytes(
127                // we need the call .to_string() to properly convert ASCII to UTF-8
128                #[allow(clippy::unnecessary_to_owned)]
129                &uuid.to_string(),
130                writer,
131            ),
132            Schema::Fixed(FixedSchema { size, .. }) => {
133                if size != 16 {
134                    return Err(Error::ConvertFixedToUuid(size));
135                }
136
137                let bytes = uuid.as_bytes();
138                encode_bytes(bytes, writer)
139            }
140            _ => Err(Error::EncodeValueAsSchemaError {
141                value_kind: ValueKind::Uuid,
142                supported_schema: vec![SchemaKind::Uuid, SchemaKind::Fixed],
143            }),
144        },
145        Value::BigDecimal(bg) => {
146            let buf: Vec<u8> = serialize_big_decimal(bg)?;
147            writer.write(buf.as_slice()).map_err(Error::WriteBytes)
148        }
149        Value::Bytes(bytes) => match *schema {
150            Schema::Bytes => encode_bytes(bytes, writer),
151            Schema::Fixed { .. } => writer.write(bytes.as_slice()).map_err(Error::WriteBytes),
152            _ => Err(Error::EncodeValueAsSchemaError {
153                value_kind: ValueKind::Bytes,
154                supported_schema: vec![SchemaKind::Bytes, SchemaKind::Fixed],
155            }),
156        },
157        Value::String(s) => match *schema {
158            Schema::String | Schema::Uuid => encode_bytes(s, writer),
159            Schema::Enum(EnumSchema { ref symbols, .. }) => {
160                if let Some(index) = symbols.iter().position(|item| item == s) {
161                    encode_int(index as i32, writer)
162                } else {
163                    error!("Invalid symbol string {:?}.", &s[..]);
164                    Err(Error::GetEnumSymbol(s.clone()))
165                }
166            }
167            _ => Err(Error::EncodeValueAsSchemaError {
168                value_kind: ValueKind::String,
169                supported_schema: vec![SchemaKind::String, SchemaKind::Enum],
170            }),
171        },
172        Value::Fixed(_, bytes) => writer.write(bytes.as_slice()).map_err(Error::WriteBytes),
173        Value::Enum(i, _) => encode_int(*i as i32, writer),
174        Value::Union(idx, item) => {
175            if let Schema::Union(ref inner) = *schema {
176                let inner_schema = inner
177                    .schemas
178                    .get(*idx as usize)
179                    .expect("Invalid Union validation occurred");
180                encode_long(*idx as i64, &mut *writer)?;
181                encode_internal(item, inner_schema, names, enclosing_namespace, &mut *writer)
182            } else {
183                error!("invalid schema type for Union: {schema:?}");
184                Err(Error::EncodeValueAsSchemaError {
185                    value_kind: ValueKind::Union,
186                    supported_schema: vec![SchemaKind::Union],
187                })
188            }
189        }
190        Value::Array(items) => {
191            if let Schema::Array(ref inner) = *schema {
192                if !items.is_empty() {
193                    encode_long(items.len() as i64, &mut *writer)?;
194                    for item in items.iter() {
195                        encode_internal(
196                            item,
197                            &inner.items,
198                            names,
199                            enclosing_namespace,
200                            &mut *writer,
201                        )?;
202                    }
203                }
204                writer.write(&[0u8]).map_err(Error::WriteBytes)
205            } else {
206                error!("invalid schema type for Array: {schema:?}");
207                Err(Error::EncodeValueAsSchemaError {
208                    value_kind: ValueKind::Array,
209                    supported_schema: vec![SchemaKind::Array],
210                })
211            }
212        }
213        Value::Map(items) => {
214            if let Schema::Map(ref inner) = *schema {
215                if !items.is_empty() {
216                    encode_long(items.len() as i64, &mut *writer)?;
217                    for (key, value) in items {
218                        encode_bytes(key, &mut *writer)?;
219                        encode_internal(
220                            value,
221                            &inner.types,
222                            names,
223                            enclosing_namespace,
224                            &mut *writer,
225                        )?;
226                    }
227                }
228                writer.write(&[0u8]).map_err(Error::WriteBytes)
229            } else {
230                error!("invalid schema type for Map: {schema:?}");
231                Err(Error::EncodeValueAsSchemaError {
232                    value_kind: ValueKind::Map,
233                    supported_schema: vec![SchemaKind::Map],
234                })
235            }
236        }
237        Value::Record(value_fields) => {
238            if let Schema::Record(RecordSchema {
239                ref name,
240                fields: ref schema_fields,
241                ..
242            }) = *schema
243            {
244                let record_namespace = name.fully_qualified_name(enclosing_namespace).namespace;
245
246                let mut lookup = HashMap::new();
247                value_fields.iter().for_each(|(name, field)| {
248                    lookup.insert(name, field);
249                });
250
251                let mut written_bytes = 0;
252                for schema_field in schema_fields.iter() {
253                    let name = &schema_field.name;
254                    let value_opt = lookup.get(name).or_else(|| {
255                        if let Some(aliases) = &schema_field.aliases {
256                            aliases.iter().find_map(|alias| lookup.get(alias))
257                        } else {
258                            None
259                        }
260                    });
261
262                    if let Some(value) = value_opt {
263                        written_bytes += encode_internal(
264                            value,
265                            &schema_field.schema,
266                            names,
267                            &record_namespace,
268                            writer,
269                        )?;
270                    } else {
271                        return Err(Error::NoEntryInLookupTable(
272                            name.clone(),
273                            format!("{lookup:?}"),
274                        ));
275                    }
276                }
277                Ok(written_bytes)
278            } else if let Schema::Union(UnionSchema { schemas, .. }) = schema {
279                let mut union_buffer: Vec<u8> = Vec::new();
280                for (index, schema) in schemas.iter().enumerate() {
281                    encode_long(index as i64, &mut union_buffer)?;
282                    let encode_res = encode_internal(
283                        value,
284                        schema,
285                        names,
286                        enclosing_namespace,
287                        &mut union_buffer,
288                    );
289                    match encode_res {
290                        Ok(_) => {
291                            return writer
292                                .write(union_buffer.as_slice())
293                                .map_err(Error::WriteBytes);
294                        }
295                        Err(_) => {
296                            union_buffer.clear(); //undo any partial encoding
297                        }
298                    }
299                }
300                return Err(Error::EncodeValueAsSchemaError {
301                    value_kind: ValueKind::Record,
302                    supported_schema: vec![SchemaKind::Record, SchemaKind::Union],
303                });
304            } else {
305                error!("invalid schema type for Record: {schema:?}");
306                return Err(Error::EncodeValueAsSchemaError {
307                    value_kind: ValueKind::Record,
308                    supported_schema: vec![SchemaKind::Record, SchemaKind::Union],
309                });
310            }
311        }
312    }
313}

315pub fn encode_to_vec(value: &Value, schema: &Schema) -> AvroResult<Vec<u8>> {
316    let mut buffer = Vec::new();
317    encode(value, schema, &mut buffer)?;
318    Ok(buffer)
319}
